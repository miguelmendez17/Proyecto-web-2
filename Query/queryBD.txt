drop table usuarios
create table usuarios
(
	nombre		varchar(30)	not null,
	apellido	varchar(30)	not null,
	edad 		int 		not null,
	nombreDeUsuario	varchar(30)	not null,
	contrasenna	varchar(30)	not null,
	fechaRegistro 	date		null,
	horaRegistro 	time		null,
	constraint 	pk_nombre_usuario	primary key (nombreDeUsuario)
);

drop table TEAMS
create table TEAMS (
	Country		varchar(30)	not null,
	Confederation	varchar(30)	not null,
	Points		int	not null,
	activated	boolean
);

drop table Flags
create table flags (
	name		varchar(100)	not null,
	flag	varchar(10000)	not null

);

drop table confederations
create table confederations(
	id		varchar(100)	not null,
	confederation	varchar(100)	not null,
	slot		int	not null,
	slotReserv	int	not null
);

FROM TEAMS

INNER JOIN confederations
